# sqlc post dynamic table

generate a dynamic table along with sqlc static table.
this binary will scan \*.sql.go modify it to support dynamic table.
sql file that has dynaTable [cashflow,cash_inex,cash_credit]
will be replace it with array of its corresponsive value.

1. add a sqlc_dyna.json at root of dyna binary

```tree
dyna
sqlc_dyna.json
```

```json
{
  "sqlPath": "/home/ubuntu/projects/go/src/sqlc_post_plugin/db/sqlc",
  "dynaTable": {
    "cashflow": ["income", "expense", "receivable", "payable"],
    "cash_inex": ["income", "expense"],
    "cash_credit": ["receivable", "payable"],
    "ref_cash": ["income", "expense", "receivable", "payable"],
    "to_cashflow": ["income", "expense", "receivable", "payable"]
  },
  "name": {
    "fileTypeName": "extDynaSplitType.go",
    "guard": "guard",
    "dynaTable": "dynamicTb"
  }
}
```

```sql
-- name: AddCashflow :one
  insert into cashflow(
      id,name --,total
  )select id,name from to_cashflow as c
  where c.id = $1
  returning id;
```

## sqlc generate output

```go
// Code generated by sqlc. DO NOT EDIT.
// versions:
//   sqlc v1.18.0
// source: authors.sql

package sqlc

import (
	"context"
)

const addCashflow = `-- name: AddCashflow :one
  insert into cashflow(
      id,name --,total
  )select id,name from to_cashflow as c
  where c.id = $1
  returning id as newID
`

func (q *Queries) AddCashflow(ctx context.Context, id int64) (int64, error) {
	row := q.db.QueryRowContext(ctx, addCashflow, id)
	var newid int64
	err := row.Scan(&newid)
	return newid, err
}
```

## after run binary

```shell
./dyna
```

this binary will

1. create extend file type call extDynaSplitType.go
   this will be used as guard fn to constrain your dynamic table

### extDynaSplitType.go

```go

package sqlc

type tTbGuard map[string]bool

var (
	dynamicTbGuard   = []string{"cash_credit", "cash_inex", "cashflow", "ref_cash", "to_cashflow"}
	cash_creditGuard = tTbGuard{"receivable": true, "payable": true}
	cash_inexGuard   = tTbGuard{"income": true, "expense": true}
	cashflowGuard    = tTbGuard{"income": true, "expense": true, "receivable": true, "payable": true}
	ref_cashGuard    = tTbGuard{"income": true, "expense": true, "receivable": true, "payable": true}
	to_cashflowGuard = tTbGuard{"income": true, "expense": true, "receivable": true, "payable": true}
)
```

2. modify any sqlc generate file that has
   dynamice table

```go
// Code generated by sqlc. DO NOT EDIT.
// versions:
//   sqlc v1.18.0
// source: authors.sql

package sqlc

import (
	"context"
	"errors"
)

type AddCashflowTbGuard struct {
	Cashflow   string
	ToCashflow string
}

func getAddCashflowQuery(tb AddCashflowTbGuard) (string, error) {
	if !to_cashflowGuard[tb.ToCashflow] ||
		!cashflowGuard[tb.Cashflow] {
		return "", errors.New(tb.ToCashflow + ", " + tb.Cashflow + " !not allow.")
	}

	return `
  insert into ` + tb.Cashflow + `(
      id,name
  )select id,name from ` + tb.ToCashflow + ` as c
  where c.id = $1
  returning id as newID
`, nil
}

func (q *Queries) AddCashflow(ctx context.Context, dynaTable AddCashflowTbGuard, id int64) (int64, error) {
	dynaQuery, errQuery := getAddCashflowQuery(dynaTable)
	if errQuery != nil {
		return 0, errQuery
	}

	row := q.db.QueryRowContext(ctx, dynaQuery, id)
	var newid int64
	err := row.Scan(&newid)
	return newid, err
}
```

now AddCashflow fun support dynamic table.
